Multi-Agent Deep Q-Network (DQN)

A complete implementation of multi-agent reinforcement learning using DQN. This code demonstrates how multiple agents can learn to interact in shared environments.

Full Implementation: example-algorithm.py

Key Components:

1. Q-Network Architecture

class QNetwork(nn.Module):
    """Neural network for approximating Q-values in multi-agent setting."""
    def __init__(self, state_dim, action_dim, hidden_dim=128):
        super(QNetwork, self).__init__()
        self.network = nn.Sequential(
            nn.Linear(state_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, action_dim)
        )

    def forward(self, state):
        return self.network(state)

2. Experience Replay Buffer

class ReplayBuffer:
    """Experience replay buffer for off-policy learning."""
    def __init__(self, capacity=10000):
        self.buffer = deque(maxlen=capacity)

    def push(self, state, action, reward, next_state, done):
        self.buffer.append((state, action, reward, next_state, done))

    def sample(self, batch_size):
        batch = random.sample(self.buffer, batch_size)
        states, actions, rewards, next_states, dones = zip(*batch)
        return (np.array(states), np.array(actions),
                np.array(rewards), np.array(next_states),
                np.array(dones))

3. Multi-Agent Coordination

def select_actions(self, states):
    """Epsilon-greedy action selection for all agents."""
    actions = []
    for i, state in enumerate(states):
        if random.random() < self.epsilon:
            action = random.randint(0, self.action_dim - 1)
        else:
            with torch.no_grad():
                state_tensor = torch.FloatTensor(state).unsqueeze(0)
                q_values = self.q_networks[i](state_tensor)
                action = q_values.argmax().item()
        actions.append(action)
    return actions
